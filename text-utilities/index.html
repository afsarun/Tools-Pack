Text Utilities Basics

What this tool can do

  • Change case: UPPER • lower • Title Case • Sentence case
  • Clean text: trim, collapse spaces, remove extra blank lines
  • Lines: sort (A→Z/Z→A), unique, dedupe, prefix/suffix, numbering
  • Counts: characters, words, lines (Unicode-aware)
  • Transform: slugify, remove diacritics, reverse text, smart quotes ←→ straight
  • Find & replace: plain or regex
  • Copy/Download results (local-only)

Runs entirely in your browser. No uploads or tracking.

Inputs

  • Paste or type text (preserves newlines)
  • Options: case mode, whitespace rules, sort/unique, regex flags
  • Output: live preview + copy/download

Unicode

  • Works with emoji and non-Latin scripts
  • Counts use Intl.Segmenter when available for better word/grapheme accuracy

Common recipes

  • Blog titles: Title Case + keep minor words lowercase (of, the, a…)
  • CSV lists: one item per line → unique → sort
  • URLs: slugify + lowercase + hyphens

How to use

  1. Paste text into the Input area.
  2. Select a utility (e.g., Title Case, Slugify, Sort/Unique).
  3. Adjust options (e.g., ignore case while deduping, regex flags for replace).
  4. See the result in Output; click Copy or Download.

Tips

  • Enable “Collapse whitespace” to normalize spaces and blank lines.
  • Use Regex replace for patterns (e.g., \\s+ → single space).
  • For languages with diacritics, apply Remove diacritics before Slugify.

Technical Implementation

Case transforms

function toUpper(s){ return s.toUpperCase(); }
function toLower(s){ return s.toLowerCase(); }

const MINOR = new Set(["a","an","and","as","at","but","by","for","in","nor","of","on","or","per","the","to","vs","via"]);
function toTitleCase(s){
  return s.split(/(\s+|-|:)/).map((tok, i, arr) => {
    if (/^\s+$/.test(tok) || tok === "-" || tok === ":") return tok;
    const lower = tok.toLowerCase();
    const isEdge = (i === 0 || i === arr.length - 1);
    if (!isEdge && MINOR.has(lower)) return lower;
    return lower.replace(/^[\p{Letter}\p{Number}]/u, ch => ch.toUpperCase());
  }).join("");
}
function toSentenceCase(s){
  return s.replace(/(^|[.!?]\s+)(\p{Letter})/gu, (m, p1, p2) => p1 + p2.toUpperCase());
}

Whitespace normalization

function trimLines(s){ return s.replace(/[ \t]+$/gm, "").replace(/^\s+|\s+$/g, ""); }
function collapseSpaces(s){ return s.replace(/\s+/g, " ").trim(); }
function collapseBlankLines(s){ return s.replace(/\n{3,}/g, "\n\n"); }

Lines: sort & unique

function lines(s){ return s.replace(/\r\n/g,"\n").split("\n"); }
function joinLines(a){ return a.join("\n"); }

function sortLinesAsc(s){ return joinLines(lines(s).sort((a,b)=>a.localeCompare(b))); }
function sortLinesDesc(s){ return joinLines(lines(s).sort((a,b)=>b.localeCompare(a))); }

function uniqueLines(s, { caseInsensitive=false, trim=true } = {}){
  const seen = new Set();
  const out = [];
  for (let ln of lines(s)) {
    const key = (caseInsensitive ? ln.toLowerCase() : ln).trim();
    const k = trim ? key : (caseInsensitive ? ln.toLowerCase() : ln);
    if (!seen.has(k)) { seen.add(k); out.push(ln); }
  }
  return joinLines(out);
}

Slugify & diacritics

function stripDiacritics(s){
  return s.normalize("NFKD").replace(/[\u0300-\u036f]/g, "");
}
function slugify(s){
  return stripDiacritics(s)
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-+|-+$/g, "")
    .slice(0, 120);
}

Smart quotes ←→ straight quotes

function toStraightQuotes(s){
  return s
    .replace(/[“”]/g, '"')
    .replace(/[‘’]/g, "'")
    .replace(/—/g, "--");
}
function toSmartQuotes(s){
  return s
    .replace(/\b"/g, "“").replace(/"\b/g, "”")
    .replace(/\b'/g, "‘").replace(/'\b/g, "’")
    .replace(/--/g, "—");
}

Find & replace (plain or regex)

function replaceAllSafe(s, search, repl, { regex=false, flags="g" } = {}){
  if (regex) {
    const rx = new RegExp(search, flags.includes("g") ? flags : flags + "g");
    return s.replace(rx, repl);
  }
  const esc = search.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
  return s.replace(new RegExp(esc, "g"), repl);
}

Accurate counts (Unicode-aware)

function countStats(s){
  const linesCount = s ? s.replace(/\r\n/g, "\n").split("\n").length : 0;
  const words = (typeof Intl !== "undefined" && Intl.Segmenter)
    ? [...new Intl.Segmenter(undefined, { granularity: "word" }).segment(s)]
        .filter(seg => seg.isWordLike).length
    : (s.trim().match(/\S+/g) || []).length;
  const chars = [...s].length; // code points (not bytes)
  return { lines: linesCount, words, characters: chars };
}

Reverse text (code-point safe)

function reverseText(s){ return [...s].reverse().join(""); }

Copy/Download helpers

async function copyText(text){
  try { await navigator.clipboard.writeText(text); return true; }
  catch {
    const ta = document.createElement("textarea");
    ta.value = text; document.body.appendChild(ta);
    ta.select(); document.execCommand("copy"); ta.remove();
    return true;
  }
}
function downloadText(text, name="text.txt"){
  const blob = new Blob([text], { type: "text/plain;charset=utf-8" });
  const url = URL.createObjectURL(blob);
  const a = Object.assign(document.createElement("a"), { href: url, download: name });
  a.click(); URL.revokeObjectURL(url);
}
