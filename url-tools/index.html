URL Utilities Basics

What this tool can do

  • Encode / Decode full URLs or individual components
  • Parse URLs into protocol, host, path, query, hash
  • Edit query params: add, update, remove (UTM builder/stripper)
  • Normalize URLs: lowercase host, remove default ports, trim dots/slashes
  • Compare origins (same-origin / same-site)

Everything runs locally in your browser—no data leaves your device.

RFC 3986 quick notes

  • Unreserved: A–Z a–z 0–9 - . _ ~ (never percent-encoded)
  • Reserved: ! * ' ( ) ; : @ & = + $ , / ? # [ ] (context-dependent)
  • Spaces: in URLs → %20; in form encoding (application/x-www-form-urlencoded) → +

Encode APIs

  • encodeURI() → for the whole URL (does not escape :/#?&)
  • encodeURIComponent() → for parts (query values, path segments)
  • Decoding uses decodeURI() / decodeURIComponent()

IDN & Punycode

  • Unicode domains (e.g., https://bücher.example) are converted to punycode internally.
  • Use the URL constructor to normalize; it serializes to ASCII when needed.

How to use

  1. Paste a URL or component string.
  2. Choose Encode / Decode or switch to Query Editor.
  3. Add / edit / remove query parameters (e.g., UTM tags) and copy the updated URL.
  4. Use Normalize to tidy hosts, protocols, trailing slashes, default ports.

Tips

  • Encode components (path segments, query values) with encodeURIComponent.
  • Avoid double-encoding. Decode once, mutate, then re-encode only components you changed.
  • For forms, URLSearchParams handles + vs %20 details for you.

URL Best Practices

🧩 Encode the right layer

Use encodeURI for the whole URL and encodeURIComponent for parts.

🧹 Normalize before compare

Lowercase host, drop default ports (:80/:443), remove trailing slashes.

🚫 Strip trackers

Remove utm_*, fbclid, gclid unless needed.

🔐 Beware secrets in URLs

Never place credentials or tokens in query strings; use headers or POST bodies.

Technical Implementation

Safe component encoding

function encodeComponentSafe(s) {
  return encodeURIComponent(String(s))
    // keep commonly safe characters if desired (optional):
    .replace(/[!'()*]/g, c => "%" + c.charCodeAt(0).toString(16).toUpperCase());
}
function decodeComponentSafe(s) {
  try { return decodeURIComponent(s); }
  catch { return s; } // fallback if malformed
}

Parse a URL into parts

function parseUrl(input) {
  const u = new URL(input);
  return {
    href: u.href,
    protocol: u.protocol, // "https:"
    username: u.username,
    password: u.password,
    host: u.host,         // "example.com:443"
    hostname: u.hostname, // "example.com"
    port: u.port,         // "443"
    origin: u.origin,     // "https://example.com"
    pathname: u.pathname, // "/a/b"
    search: u.search,     // "?q=1"
    hash: u.hash          // "#top"
  };
}

Build & mutate query params

function setParams(url, obj = {}) {
  const u = new URL(url);
  const p = u.searchParams;
  Object.entries(obj).forEach(([k, v]) => {
    if (v === null || v === undefined) p.delete(k);
    else p.set(k, String(v));
  });
  u.search = p.toString(); // serializes with proper encoding
  return u.toString();
}

function getParams(url) {
  const p = new URL(url).searchParams;
  const out = {};
  p.forEach((v, k) => { out[k] = v; });
  return out;
}

UTM builder & stripper

function addUTM(url, { source, medium, campaign, term, content } = {}) {
  const u = new URL(url);
  if (source)  u.searchParams.set("utm_source", source);
  if (medium)  u.searchParams.set("utm_medium", medium);
  if (campaign)u.searchParams.set("utm_campaign", campaign);
  if (term)    u.searchParams.set("utm_term", term);
  if (content) u.searchParams.set("utm_content", content);
  return u.toString();
}
function stripTracking(url) {
  const u = new URL(url);
  const junk = [
    /^utm_/i, /^fbclid$/i, /^gclid$/i, /^dclid$/i,
    /^mc_(cid|eid)$/i, /^igshid$/i, /^vero_id$/i
  ];
  [...u.searchParams.keys()].forEach(k => {
    if (junk.some(rx => rx.test(k))) u.searchParams.delete(k);
  });
  return u.toString();
}

Normalize URL (host, ports, slashes)

function normalizeUrl(url) {
  const u = new URL(url);
  u.hostname = u.hostname.toLowerCase();
  // drop default ports
  if ((u.protocol === "http:" && u.port === "80") ||
      (u.protocol === "https:" && u.port === "443")) {
    u.port = "";
  }
  // collapse multiple slashes in path (except after protocol)
  u.pathname = u.pathname.replace(/\/{2,}/g, "/");
  // remove trailing slash for non-root
  if (u.pathname.length > 1 && u.pathname.endsWith("/")) {
    u.pathname = u.pathname.slice(0, -1);
  }
  return u.toString();
}

Same-origin / same-site checks

function sameOrigin(a, b) {
  const A = new URL(a), B = new URL(b);
  return A.origin === B.origin;
}
function sameSite(a, b) {
  const A = new URL(a), B = new URL(b);
  const etld1 = A.hostname.split(".").slice(-2).join(".");
  const etld2 = B.hostname.split(".").slice(-2).join(".");
  return etld1 === etld2 && A.protocol === B.protocol;
}

Form-style encode/decode (space ↔ +)

function formEncode(params) {
  const usp = new URLSearchParams(params);
  return usp.toString(); // uses '+' for spaces
}
function formDecode(qs) {
  const p = new URLSearchParams(qs.startsWith("?") ? qs.slice(1) : qs);
  const out = {};
  for (const [k, v] of p.entries()) out[k] = v;
  return out;
}

Path segment helpers

function joinPath(...segs) {
  return segs
    .map(s => String(s).replace(/^\/+|\/+$/g, ""))
    .filter(Boolean)
    .join("/")
    .replace(/^/, "/");
}
// Example: new URL(joinPath("api", "v1", "users"), "https://example.com").toString();
