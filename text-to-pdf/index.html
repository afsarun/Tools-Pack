Text to PDF Basics

What this tool can do

  • Convert plain text to a paginated PDF
  • Page setup: A4/Letter, portrait/landscape, margins
  • Typography: font family/size/line-height, left/center/right align
  • Headers/Footers: optional title, page numbers
  • Unicode support (e.g., Bangla, emoji) with embedded fonts

Everything runs locally in your browser‚Äîyour text never leaves your device.

Inputs

  • Text: paste or type; preserves line breaks
  • Page: A4 or Letter; portrait/landscape
  • Layout: margins (e.g., 36‚Äì72 pt), font size/line-height
  • Extras: title header, page numbers, filename

Output characteristics

  • PDF 1.x file with embedded fonts (when provided)
  • Hard wraps at word boundaries; automatic pagination
  • Consistent metrics via monospaced or measured proportional fonts

Unicode & fonts

  • To render non-Latin text (e.g., ‡¶¨‡¶æ‡¶Ç‡¶≤‡¶æ), embed a Unicode TTF like Noto Sans.
  • Without an embedded font, some libraries fall back to limited built-ins (may miss glyphs).
  • Embed once; reuse for all pages to keep file size reasonable.

How to use

  1. Paste your text into the editor.
  2. Choose page size, margins, and font options.
  3. (Optional) Add a title and enable page numbers.
  4. Click Generate PDF, then Download.

Tips

  • Use a monospace font for code or logs to keep alignment.
  • For long documents, set line-height to ~1.4 for readability.
  • If characters appear as boxes, embed a Unicode TTF (e.g., Noto Sans).

Technical Implementation

Option A ‚Äî jsPDF (CDN)

<script src="https://cdn.jsdelivr.net/npm/jspdf@2.5.1/dist/jspdf.umd.min.js"></script>
async function textToPdf_jsPDF({
  text,
  title = "",
  filename = "document.pdf",
  unit = "pt",
  format = "a4",           // "a4" | "letter" | [w,h]
  orientation = "portrait",
  margins = { top: 54, right: 54, bottom: 54, left: 54 },
  fontSize = 12,
  lineHeight = 1.4,
  align = "left",          // "left" | "center" | "right"
  embedFont = null,        // { name, normalTTFBase64 } optional
  pageNumbers = true
} = {}) {
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF({ unit, format, orientation });

  // Embed Unicode font (optional)
  if (embedFont?.normalTTFBase64) {
    const vfsName = `${embedFont.name}-normal.ttf`;
    doc.addFileToVFS(vfsName, embedFont.normalTTFBase64);
    doc.addFont(vfsName, embedFont.name, "normal");
    doc.setFont(embedFont.name, "normal");
  }

  const pageW = doc.internal.pageSize.getWidth();
  const pageH = doc.internal.pageSize.getHeight();
  const contentW = pageW - margins.left - margins.right;
  const startX = margins.left;
  let cursorY = margins.top;

  doc.setFontSize(fontSize);

  // Header (title)
  if (title) {
    const titleLines = doc.splitTextToSize(title, contentW);
    doc.text(titleLines, startX, cursorY);
    cursorY += fontSize * lineHeight * titleLines.length + 8;
  }

  // Split text into wrapped lines
  const lines = doc.splitTextToSize(text.replace(/\r\n/g, "\n"), contentW);
  const lineStep = fontSize * lineHeight;

  const footer = (page, pages) => {
    if (!pageNumbers) return;
    const label = `Page ${page} of ${pages}`;
    doc.setFontSize(10);
    const y = pageH - margins.bottom / 2;
    const x = pageW / 2;
    doc.text(label, x, y, { align: "center" });
    doc.setFontSize(fontSize);
  };

  // Render lines with pagination
  let page = 1;
  const drawLine = (txt, y) => {
    const opts = align === "left" ? {} : { align, maxWidth: contentW };
    const x = align === "center" ? pageW / 2 : (align === "right" ? pageW - margins.right : startX);
    doc.text(txt, x, y, opts);
  };

  const maxY = () => pageH - margins.bottom;
  for (let i = 0; i < lines.length; i++) {
    if (cursorY + lineStep > maxY()) {
      footer(page, undefined); // total pages unknown until end; optional skip
      doc.addPage();
      page++;
      cursorY = margins.top;
    }
    drawLine(lines[i], cursorY);
    cursorY += lineStep;
  }

  // Back-fill total pages (simple approach: add footer only on last page)
  footer(page, page);

  doc.save(filename);
}

Note: To fully number ‚ÄúPage X of Y‚Äù, render after measuring pages or use a two-pass strategy (render ‚Üí count ‚Üí add footers).

Option B ‚Äî pdf-lib (font embedding, fine control)

<script src="https://cdn.jsdelivr.net/npm/pdf-lib/dist/pdf-lib.min.js"></script>
async function textToPdf_pdfLib({
  text,
  title = "",
  filename = "document.pdf",
  pageSize = "A4",         // "A4" | "Letter"
  orientation = "portrait",
  margins = { top: 54, right: 54, bottom: 54, left: 54 },
  fontSize = 12,
  lineHeight = 1.4,
  align = "left",
  fontBytes,               // Uint8Array (e.g., NotoSans-Regular.ttf)
  pageNumbers = true
} = {}) {
  const { PDFDocument, StandardFonts } = PDFLib;
  const doc = await PDFDocument.create();

  // Size
  const sizes = {
    A4: [595.28, 841.89],
    Letter: [612, 792]
  };
  let [w, h] = sizes[pageSize] || sizes.A4;
  if (orientation === "landscape") [w, h] = [h, w];

  // Font
  const font = fontBytes
    ? await doc.embedFont(fontBytes, { subset: true })
    : await doc.embedFont(StandardFonts.Helvetica);

  const lineStep = fontSize * lineHeight;

  // Word wrap using font width
  function wrap(text, maxWidth) {
    const out = [];
    const paragraphs = text.replace(/\r\n/g, "\n").split("\n");
    for (const p of paragraphs) {
      let line = "";
      for (const word of p.split(/\s+/)) {
        const test = line ? line + " " + word : word;
        const width = font.widthOfTextAtSize(test, fontSize);
        if (width <= maxWidth) {
          line = test;
        } else {
          if (line) out.push(line);
          // very long token fallback
          if (font.widthOfTextAtSize(word, fontSize) > maxWidth) {
            // hard-break the word
            let chunk = "";
            for (const ch of word) {
              const test2 = chunk + ch;
              if (font.widthOfTextAtSize(test2, fontSize) > maxWidth) {
                out.push(chunk);
                chunk = ch;
              } else {
                chunk = test2;
              }
            }
            line = chunk;
          } else {
            line = word;
          }
        }
      }
      out.push(line);
    }
    return out;
  }

  const page = doc.addPage([w, h]);
  const contentW = w - margins.left - margins.right;
  let y = h - margins.top;

  // Title
  if (title) {
    const titleLines = wrap(title, contentW);
    for (const tl of titleLines) {
      const tw = font.widthOfTextAtSize(tl, fontSize);
      const x =
        align === "center" ? (w - tw) / 2 :
        align === "right"  ? w - margins.right - tw :
                             margins.left;
      page.drawText(tl, { x, y, size: fontSize, font });
      y -= lineStep;
    }
    y -= 8;
  }

  // Body
  const lines = wrap(text, contentW);

  let current = page;
  let pageIndex = 1;
  const pages = [current];

  for (const ln of lines) {
    if (y - lineStep < margins.bottom) {
      current = doc.addPage([w, h]);
      pages.push(current);
      y = h - margins.top;
      pageIndex++;
    }
    const tw = font.widthOfTextAtSize(ln, fontSize);
    const x =
      align === "center" ? (w - tw) / 2 :
      align === "right"  ? w - margins.right - tw :
                           margins.left;
    current.drawText(ln, { x, y, size: fontSize, font });
    y -= lineStep;
  }

  // Footers
  if (pageNumbers) {
    const total = pages.length;
    for (let i = 0; i < total; i++) {
      const p = pages[i];
      const label = `Page ${i + 1} of ${total}`;
      const tw = font.widthOfTextAtSize(label, 10);
      p.drawText(label, {
        size: 10,
        font,
        x: (w - tw) / 2,
        y: margins.bottom / 2
      });
    }
  }

  const bytes = await doc.save();
  const blob = new Blob([bytes], { type: "application/pdf" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url; a.download = filename; a.click();
  URL.revokeObjectURL(url);
}
